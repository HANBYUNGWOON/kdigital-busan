CREATE TABLE BOARD 
(	
	BOARDNO INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	TITLE VARCHAR(100) NOT NULL, 
	WRITER VARCHAR(20) NOT NULL, 
	CONTENT VARCHAR(4000) NOT NULL, 
	REGDATE DATETIME DEFAULT (NOW()), 
	READCOUNT INT DEFAULT (0), 
	DELETED BOOLEAN DEFAULT (FALSE),
	CONSTRAINT FK_MEMBER_BOARD FOREIGN KEY (WRITER) REFERENCES MEMBER(MEMBERID)
);
  
CREATE TABLE BOARDATTACH 
(	
	ATTACHNO INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	BOARDNO INT NOT NULL, 
	SAVEDFILENAME VARCHAR(100) NOT NULL, 
	USERFILENAME VARCHAR(100) NOT NULL,	
	CONSTRAINT FK_BOARD_ATTACH FOREIGN KEY (BOARDNO) REFERENCES BOARD(BOARDNO)
);
  
CREATE TABLE BOARDCOMMENT 
(	
	COMMENTNO INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	BOARDNO INT NOT NULL, 
	WRITER VARCHAR(20) NOT NULL, 
	CONTENT VARCHAR(500) NOT NULL, 
	REGDATE DATETIME DEFAULT (NOW()), 
	GROUPNO INT NOT NULL, 
	STEP INT NOT NULL, 
	DEPTH INT DEFAULT (0), 
	DELETED BOOLEAN DEFAULT (FALSE),
	CONSTRAINT FK_BOARD_COMMENT FOREIGN KEY (BOARDNO) REFERENCES BOARD(BOARDNO),
	CONSTRAINT FK_MEMBER_COMMENT FOREIGN KEY (WRITER) REFERENCES MEMBER(MEMBERID)
);
